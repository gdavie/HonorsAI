168  public static void MarkIntrons(ArrayList<Instruction> instructions, int numUsedRegisters){
169  /*
170  Source: Brameier, M 2004 On Linear Genetic Programming (thesis)
171 
172  Algorithm 3.1 (detection of structural introns)
173  1. Let set R_eff always contain all registers that are effective at the current program
174  position. R_eff := { r | r is output register }.
175  Start at the last program instruction and move backwards.
176  2. Mark the next preceding operation in program with:
177  destination register r_dest element-of R_eff.
178  If such an instruction is not found then go to 5.
179  3. If the operation directly follows a branch or a sequence of branches then mark these
180  instructions too. Otherwise remove r_dest from R_eff .
181  4. Insert each source (operand) register r_op of newly marked instructions in R_eff
182  if not already contained. Go to 2.
183  5. Stop. All unmarked instructions are introns.
184  */
185 
186  // Assume all instructions are introns and mark as such. We do it backwards because we need
187  // a reverse iterator later on and hell, why allocate two iterators - clutters the code and
188  // the memory.
189  for(int i = instructions.size()-1; i >=0; i--) {
190  Instruction ri = instructions.get(i);
191  ri.setIntron(true);
192  }
193 
194  // Assume all registers are involved in the output (base class can override this if need be)
195  Set<Integer> usedRegisters = new HashSet<Integer>();
196 
197  for(int i = 0; i <= numUsedRegisters; ++i) {
198  usedRegisters.add(i);
199  }
200 
201  // Now go through steps 2-4 of the algorithm:
202  for(int i = instructions.size()-1; i >= 0; --i) {
203  Instruction ri = instructions.get(i);
204  if(ri.isConditional()) {
205  // Then: If the next instruction is not an intron and if this instruction is not the last
206  // one in the program then this instruction is also not an intron, otherwise it is:
207  if(i != instructions.size()-1 && !instructions.get(i+1).isIntron()) {
208  ri.setIntron(false);
209  }
210  }
211  else { // Instruction ri is an assignment instruction
212  if(usedRegisters.contains(ri.destinationIndex())) {
213  ri.setIntron(false);
214  usedRegisters.remove(ri.destinationIndex());
215 
216  if(ri.firstArgumentType() == ArgumentType.register) {
217  usedRegisters.add(ri.firstArgumentIndex());
218  }
219 
220  if(ri.secondArgumentType() == ArgumentType.register) {
221  usedRegisters.add(ri.secondArgumentIndex());
222  }
223  }
224  }
